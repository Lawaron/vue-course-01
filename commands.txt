docker build \
    --build-arg HOST_UID=$(id -u) \
    --build-arg HOST_GID=$(id -g) \
    -t my-vue-container \
    -f $(pwd)/docker/Dockerfile .

docker run -it --rm \
    -v $(pwd)/app:/home/node/app \
    -p 80:5173 \
    my-vue-container /bin/bash

npm run dev -- --host
npm install firebase

docker-compose -f docker/docker-compose.yml up --detach --build
docker exec -it my-vue-container /bin/bash
